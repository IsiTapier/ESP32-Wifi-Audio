249  template<typename T>
250  void processSamples(const void *data, size_t byteCount) {
251  T *dataT = (T*) data;
252  T sample;
253  float sample_windowed;
254  int samples = byteCount/sizeof(T);
255  for (int j=0; j<samples; j+=cfg.channels){
256  sample = dataT[j+cfg.channel_used];
257  sample_windowed = sample;
258  // optionally apply window function
259  if (cfg.window_function!=nullptr){
260  sample_windowed = cfg.window_function->factor(current_pos) * sample;
261  }
262  p_driver->setValue(current_pos, sample_windowed);
263  writeStrideBuffer((uint8_t*)&sample, sizeof(T));
264  if (++current_pos>=cfg.length){
265  fft();
266  }
267  }
268  }

296  bool InsertSorted(AudioFFTResult(&result)[N], AudioFFTResult tmp){
297  for (int j=0;j<N;j++){
298  if (tmp.magnitude>result[j].magnitude){
299  // shift existing values right
300  for (int i=N-2;i>=j;i--){
301  result[i+1] = result[i];

305  }
