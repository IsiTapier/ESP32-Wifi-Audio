207  size_t write(const uint8_t *data, size_t len) override {
208  int open = len;
209  size_t result = 0;
210  int retry_count = 0;
211  while (open > 0) {
212  if (available_to_write > 0) {
213  resetAvailableToWrite();
214  size_t send_len = min(open, ESP_NOW_MAX_DATA_LEN);
215  esp_err_t rc = esp_now_send(nullptr, data + result, send_len);
216  // wait for confirmation
217  if (cfg.use_send_ack) {
218  while (available_to_write == 0) {
219  delay(1);
220  }
221  } else {
222  is_write_ok = true;
223  }
224  // check status
225  if (rc == ESP_OK && is_write_ok) {
226  open -= send_len;
227  result += send_len;
228  } else {
229  LOGE("Write error");
230  retry_count++;
231  if (cfg.write_retry_count>0 && retry_count>=cfg.write_retry_count){
232  // break loop
233  return 0;
234  }
235  }
236  // if we do have no partner to write we stall and retry later
237  } else {
238  delay(cfg.delay_after_write_ms);
239  }
240 
241  // Wait some time before we retry
242  if (!is_write_ok) {
243  delay(cfg.delay_after_failed_write_ms);
244  }
245  }
246  return result;
247  }
