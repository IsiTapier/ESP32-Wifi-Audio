188  // calculates a single sample using the indicated state
189  float sample(EQSTATE &es, float sample) {
190  // Locals
191  float l,m,h; // Low / Mid / High - Sample Values
192  // Filter #1 (lowpass)
193  es.f1p0 += (es.lf * (sample - es.f1p0)) + vsa;
194  es.f1p1 += (es.lf * (es.f1p0 - es.f1p1));
195  es.f1p2 += (es.lf * (es.f1p1 - es.f1p2));
196  es.f1p3 += (es.lf * (es.f1p2 - es.f1p3));
197 
198  l = es.f1p3;
199 
200  // Filter #2 (highpass)
201  es.f2p0 += (es.hf * (sample - es.f2p0)) + vsa;
202  es.f2p1 += (es.hf * (es.f2p0 - es.f2p1));
203  es.f2p2 += (es.hf * (es.f2p1 - es.f2p2));
204  es.f2p3 += (es.hf * (es.f2p2 - es.f2p3));
205 
206  h = es.sdm3 - es.f2p3;
207  // Calculate midrange (signal - (low + high))
208  m = es.sdm3 - (h + l);
209  // Scale, Combine and store
210  l *= p_cfg->gain_low;
211  m *= p_cfg->gain_medium;
212  h *= p_cfg->gain_high;
213 
214  // Shuffle history buffer
215  es.sdm3 = es.sdm2;
216  es.sdm2 = es.sdm1;
217  es.sdm1 = sample;
218 
219  // Return result
220  return(l + m + h);
221  }
