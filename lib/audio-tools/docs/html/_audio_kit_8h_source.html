69  audio_hal_iface_samples_t toSampleRate() {
70  LOGD(LOG_METHOD);
71  const static int ia[] = {8000, 11025, 16000, 22050,
72  24000, 32000, 44100, 48000};
73  const static audio_hal_iface_samples_t oa[] = {
74  AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,

76  AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
77  int diff = 99999;
78  int result = 0;
79  for (int j = 0; j < 8; j++) {
80  if (ia[j] == sample_rate) {
81  LOGD("-> %d",ia[j])
82  return oa[j];
83  } else {
84  int new_diff = abs(oa[j] - sample_rate);
85  if (new_diff < diff) {
86  result = j;
87  diff = new_diff;
88  }
89  }
90  }
91  LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
92  return oa[result];

225  virtual size_t readBytes(uint8_t *data, size_t length) override {
226  if (cfg.channels == 2) {
227  return kit.read(data, length);
228  } else if (cfg.channels==1) {
229  // convert 2 channels of int16_t to 1
230  int16_t temp[length];
231  int len_res = kit.read(temp, length*2);
232  int res_count = len_res / 2;
233  int16_t *out = (int16_t*) data;
234  for (int j=0; j<res_count; j+=2){
235  int32_t total = temp[j];
236  total+=temp[j+1];
237  total = total/2;
238  *out = total;
239  out++;
240  }
241  return res_count;
242  }
243  LOGE("Unsuported number of channels : %d", cfg.channels);
244  return 0;

249  virtual void setAudioInfo(AudioBaseInfo info) {
250  LOGI(LOG_METHOD);
251 
252  if (cfg.sample_rate != info.sample_rate
253  && cfg.bits_per_sample == info.bits_per_sample
254  && cfg.channels == info.channels
255  && is_started) {
256  // update sample rate only
257  cfg.sample_rate = info.sample_rate;
258  cfg.logInfo();
259  converter.setAudioInfo(cfg);
260  kit.setSampleRate(cfg.toSampleRate());
261  } else if (cfg.sample_rate != info.sample_rate
262  || cfg.bits_per_sample != info.bits_per_sample
263  || cfg.channels != info.channels
264  || !is_started) {
265  // more has changed and we need to start the processing
266  cfg.sample_rate = info.sample_rate;
267  cfg.bits_per_sample = info.bits_per_sample;
268  cfg.channels = info.channels;
269  cfg.logInfo();
270 
271  // Stop first
272  if(is_started){
273  kit.end();
274  }
275  // update input format
276  converter.setAudioInfo(cfg);
277  // start kit with new config
278  kit.begin(cfg.toAudioKitConfig());
279  is_started = true;
