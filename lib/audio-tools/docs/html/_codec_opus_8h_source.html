301  void encodeFrame(int lenBytes) {
302  if (lenBytes > 0) {
303  int frames = lenBytes / cfg.channels / sizeof(int16_t);
304  LOGD("opus_encode - frame_size: %d", frames);
305  int len = opus_encode(enc, (opus_int16 *)frame.data(), frames,
306  packet.data(), cfg.max_buffer_size);
307  if (len < 0) {

309  } else if (len > 0 && len <= cfg.max_buffer_size) {
