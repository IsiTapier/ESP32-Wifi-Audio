164  bool checkPrefix(const void *input_buffer, size_t length) {
165  bool result = true;
166  if (length >= 4 && memcmp(input_buffer, "\x4b\xbf\x4b\xbf", 4) == 0) {
167  if (is_hd) {
168  LOGE("aptX audio stream (not aptX HD)");
169  result = false;
170  }
171  } else if (length >= 6 &&
172  memcmp(input_buffer, "\x73\xbe\xff\x73\xbe\xff", 6) == 0) {
173  if (!is_hd) {
174  LOGE("aptX HD audio stream");
175  result = false;
176  }
177  } else {
178  if (length >= 4 && memcmp(input_buffer, "\x6b\xbf\x6b\xbf", 4) == 0) {
179  LOGE("standard aptX audio stream - not supported");
180  result = false;
181  } else {
182  LOGE("No aptX nor aptX HD audio stream");
183  result = false;
