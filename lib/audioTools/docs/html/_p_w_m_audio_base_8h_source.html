323 virtual int nextValue() {
324 int result = 0;
325 switch(audio_config.bits_per_sample ){
326 case 8: {
327 int16_t value = buffer->read();
328 if (value<0){
329 LOGE("Could not read full data");
330 value = 0;
331 }
332 result = map(value, -NumberConverter::maxValue(8), NumberConverter::maxValue(8), 0, maxOutputValue());
333 break;
334 }
335 case 16: {
336 int16_t value;
337 if (buffer->readArray((uint8_t*)&value,2)!=2){
338 LOGE("Could not read full data");
339 }
340 result = map(value, -NumberConverter::maxValue(16), NumberConverter::maxValue(16), 0, maxOutputValue());
341 break;
342 }
343 case 24: {
344 int24_t value;
345 if (buffer->readArray((uint8_t*)&value,3)!=3){
346 LOGE("Could not read full data");
347 }
348 result = map((int32_t)value, -NumberConverter::maxValue(24), NumberConverter::maxValue(24), 0, maxOutputValue());
349 break;
350 }
351 case 32: {
352 int32_t value;
353 if (buffer->readArray((uint8_t*)&value,4)!=4){
354 LOGE("Could not read full data");
355 }
356 result = map(value, -NumberConverter::maxValue(32), NumberConverter::maxValue(32), 0, maxOutputValue());
357 break;
358 }
359 }
360 return result;
361 }
