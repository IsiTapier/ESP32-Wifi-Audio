175 // find closest frequency for requested sample_rate
176 float freq_requested = cfg.sample_rate * cfg.bits_per_sample ;
177 float selected_freq = 0;
178 for (auto freq : freq_table) {
179 for (auto div : ratio_table) {
180 float freq_value = freq.freq * 1000000 / div.ratio;
181 if (abs(freq_value-freq_requested) < abs(selected_freq-freq_requested)){
182 // MCKFREQ
183 NRF_I2S->CONFIG.MCKFREQ = freq.id << I2S_CONFIG_MCKFREQ_MCKFREQ_Pos;
184 // Ratio
185 NRF_I2S->CONFIG.RATIO = div.id << I2S_CONFIG_RATIO_RATIO_Pos;
186 selected_freq = freq_value;
187 }
188 }
189 LOGI("frequency requested %f vs %f", freq_requested, selected_freq);
