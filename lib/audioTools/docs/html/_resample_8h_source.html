138 // reduces the sampes by the indicated factor.
139 size_t downsample(T *from,T *to, int sample_count, int channels, int factor ){
140 if (sample_count%factor!=0){
141 LOGE("Incompatible buffer length for down sampling. If must be a factor of %d", factor);
142 return 0;
143 }
144 int to_pos=0;
145 int frame_count = sample_count / channels;
146 size_t result = 0;
147 for (int16_t j=0; j<frame_count; j+=factor){
148 long total[channels];
149 for (int8_t ch=0; ch<channels; ch++){
150 total[ch]=0;
151 int pos = j+ch;
152 for (int16_t f=0; f<factor; f++){
153 total[ch] += *p_data(j+f, ch, from);
154 }
155 to_pos = j/factor;
156 *p_data(to_pos, ch, to) = total[ch] / factor;
157 result++;
158 }
159 }
160 return result;
161 }

164 size_t upsample(T *from, T* to, int sample_count, int channels, int factor ){
165 int frame_count = sample_count/channels;
166 size_t result = 0;
167 int pos = 0;
168 // we start with the last saved frame
169 for (int16_t frame_pos=-1; frame_pos<frame_count-1; frame_pos++){
170 for (int8_t ch=0; ch<channels; ch++){
171 T actual_data = *p_data(frame_pos, ch, from);
172 T next_data = *p_data(frame_pos+1, ch, from);
173 float diff = (next_data - actual_data) / factor;
174 pos =(frame_pos+1)*factor;
175 *p_data(pos, ch, to) = actual_data;
176 result++;
177 for (int16_t f=1;f<factor;f++){
178 pos = ((frame_pos+1)*factor)+f;
179 T tmp = actual_data + (diff*f);
180 *p_data(pos, ch, to) = tmp;
181 result++;
182 }
183 }
184 }
185 // save last frame_pos data
186 pos = frame_count*factor;
187 for (int8_t ch=0; ch<channels; ch++){
188 T tmp = *p_data(frame_count-1, ch, from);
189 last_end[ch] = tmp;
190 }
191 return pos*channels;
192 }
193
194 // provide access to data
195 T* p_data(int frame_pos, int channel, T*start){
196 return frame_pos>=0 ? start+(frame_pos*channels)+channel : last_end+channel;
197 }
