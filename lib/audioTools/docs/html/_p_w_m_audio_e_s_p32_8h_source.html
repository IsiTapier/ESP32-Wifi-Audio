80 // use sequential pins starting from start_pin
81 LOGD("-> defining pin %d",audio_config.start_pin + j);
82 pins[j].gpio = audio_config.start_pin + j;
83 } else {
84 // use defined pins
85 LOGD("-> defining pin %d",audio_config.pins[j]);
86 pins[j].gpio = audio_config.pins[j];
87 }
88 LOGD("-> ledcSetup: frequency=%d / resolution=%d", freq, audio_config.resolution);
89 ledcSetup(pwmChannel, freq, audio_config.resolution);
90 LOGD("-> ledcAttachPin: %d", pins[j].gpio);

103 virtual void setupTimer() {

105
106 // Attach timer int at sample rate

110 uint32_t counter = 40000000 / audio_config.sample_rate;
111 LOGI("-> timer counter is %zu", counter);
112 LOGD("-> timerAttachInterrupt");
113 bool interrupt_edge_type = true;
114 timerAttachInterrupt(timer, &defaultPWMAudioOutputCallback, interrupt_edge_type);
115 LOGD("-> timerAlarmWrite");
116 bool auto_reload = true;
117 timerAlarmWrite(timer, counter, auto_reload); // Timer fires at ~44100Hz [40Mhz / 907]
118 }
