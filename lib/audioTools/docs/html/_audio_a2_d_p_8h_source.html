191 virtual size_t write(const uint8_t* data, size_t len) {
192 if (a2dp_buffer==nullptr) return 0;
193 LOGD("%s: %zu", LOG_METHOD, len);
194
195 // blocking write - we wait for space in buffer
196 bool isBufferFull = true;
197 while(isBufferFull){
198 lockSemaphore(true);
199 isBufferFull = a2dp_buffer->availableForWrite()<len;
200 lockSemaphore(false);
201
202 // we wait until the buffer is full
203 if (isBufferFull){
204 if (!is_a2dp_active){
205 is_a2dp_active = true;
206 LOGW("is_a2dp_active -> true with %d bytes", a2dp_buffer->available());
207 }
208 delay(100);
209 LOGD("Waiting for free buffer space - available: %d", a2dp_buffer->available());
210 }
211 }
212
213 // write to buffer
214 lockSemaphore(true);
215 size_t result = a2dp_buffer->writeArray(data, len);
216 lockSemaphore(false);
217 LOGD("write %d -> %d", len, result);
218 return result;
219 }

310 // callback used by A2DP to provide the a2dp_source sound data
311 static int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
312 if (a2dp_buffer==nullptr) return 0;
313 int32_t result_len = 0;
314
315 bool isAvailable = false;
316 bool isLocked = false;
317 if (is_a2dp_active) {
318 if (A2DPStream_self->lockSemaphore(true)) {
319 isAvailable = a2dp_buffer->available()>0;
320 LOGD("buffer: %d, free %d",a2dp_buffer->available(), a2dp_buffer->availableForWrite() )
321 isLocked = true;
322 } else {
323 LOGE("lock failed");
324 }
325 }
326
327 // at first call we start with some empty data
328 if (isAvailable){
329 // the data in the file must be in int16 with 2 channels
330 size_t result_len_bytes = a2dp_buffer->readArray((uint8_t*)data, len*sizeof(Frame));
331 A2DPStream_self->lockSemaphore(false);
332
333 // result is in number of frames
334 result_len = result_len_bytes / sizeof(Frame);
335 } else {
336 if (isLocked) A2DPStream_self->lockSemaphore(false);
337
338 // prevent underflow on first call
339 switch (A2DPStream_self->config.noData) {
340 case A2DPSilence:
341 memset(data, 0, len*sizeof(Frame));
342 break;
343 case A2DPWhoosh:
344 for (int j=0;j<len;j++){
345 data[j].channel2 = data[j].channel1 = (rand() % 50) - 25;
346 }
347 break;
348 }
349 result_len = len;
350 delay(3);
351
352 // Priority: 22 on core 0
353 // LOGI("Priority: %d on core %d", uxTaskPriorityGet(NULL), xPortGetCoreID());
354
355 }
356
357 LOGD("a2dp_stream_source_sound_data: %d -> %d (%s)", len, result_len, isAvailable?"+":"-");
358 return result_len;
359 }
