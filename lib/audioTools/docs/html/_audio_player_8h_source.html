354 void getFileAtIndex(AudioFile dir, size_t pos, size_t& idx, AudioFile& result) {
355 LOGD("%s: %d", LOG_METHOD, idx);
356 char file_name_act[MAX_FILE_LEN];
357 dir.getName(file_name_act, MAX_FILE_LEN);
358 LOGD("-> processing directory: %s ", file_name_act);
359 AudioFile file;
360 dir.rewind();
361 while (!result && file.openNext(&dir, O_RDONLY)) {
362 // indent for dir level
363 if (!file.isHidden()) {
364 file.getName(file_name_act, MAX_FILE_LEN);
365 LOGD("-> processing: %s with index %d", file_name_act, idx);
366
367 if (isValidAudioFile(file)){
368 if (idx == pos) {
369 result = file;
370 result.getName(file_name, MAX_FILE_LEN);
371 LOGD("==> found: '%s' at index %d", file_name, idx);
372 }
373 idx++;
374 }
375
376 if (file.isDir()) {
377 getFileAtIndex(file, pos, idx, result);
378 }
379 }
380
381 if (file.dirIndex()!=result.dirIndex()){
382 LOGD("Close: %s", file_name_act);
383 file.close();
384 }
385 }
386 return;
387 }
