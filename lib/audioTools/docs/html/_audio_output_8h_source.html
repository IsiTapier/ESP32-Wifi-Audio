507 // writes all input samples
508 void write(const uint8_t *dataIn, size_t len, Print &out){
509 this->data = (T*)dataIn;
510 this->size_val = len;
511 // we process all input samples
512 for (size_t j=0;j<size();j++){
513 T value = data[j];
514 // convert to output format
515 switch(p_convert->target_bits_per_sample){
516 case 8: {
517 int8_t out_value8 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
518 byte arr[p_convert->multiply];
519 memset(arr, out_value8, p_convert->multiply);
520 out.write((const uint8_t*)arr, p_convert->multiply);
521 } break;
522 case 16: {
523 int16_t out_value16 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
524 for (int ch=0;ch<p_convert->multiply;ch++){
525 out.write((const uint8_t*)&out_value16, 2);
526 }
527 } break;
528 case 24: {
529 int32_t out_value24 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
530 for (int ch=0;ch<p_convert->multiply;ch++){
531 out.write((const uint8_t*)&out_value24, 3);
532 }
533 } break;
534 case 32: {
535 int32_t out_value32 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
536 for (int ch=0;ch<p_convert->multiply;ch++){
537 out.write((const uint8_t*)&out_value32, 4);
538 }
539 } break;
540 }
541 }
542 }

595 virtual size_t write(const uint8_t *data, size_t len){
596 size_t result = 0;
597 if (no_conversion){
598 result = p_out->write(data, len);
599 } else if (convert.multiply>0){
600 switch(p_info_in->bits_per_sample) {
601 case 8: {
602 array8.write(data, len, *p_out);
603 result = len;
604 } break;
605 case 16: {
606 array16.write(data, len/2, *p_out);
607 result = len;
608 } break;
609 case 24: {
610 array24.write(data, len/3,*p_out);
611 result = len;

623 bool no_conversion = false;
624 Print *p_out;

626 Convert convert;
627 AudioArray<int8_t> array8 = AudioArray<int8_t>(convert);
628 AudioArray<int16_t> array16 = AudioArray<int16_t>(convert);
629 AudioArray<int24_t> array24 = AudioArray<int24_t>(convert);

633 void setup() {
634 if (p_info_out!=nullptr && p_info_in!=nullptr){
635 LOGI(LOG_METHOD);
636 convert.source_bits_per_sample = p_info_in->bits_per_sample;
637 convert.target_bits_per_sample = p_info_out->bits_per_sample;
638 if (convert.source_bits_per_sample != convert.target_bits_per_sample){
639 convert.format_factor = NumberConverter::maxValue(convert.target_bits_per_sample) / NumberConverter::maxValue(convert.source_bits_per_sample);
640 LOGI("FormatConverterStream: bits %d -> %d - factor %f",convert.source_bits_per_sample, convert.target_bits_per_sample, convert.format_factor);
641 } else {
642 convert.format_factor = 1.0;
643 }
644
645 if (p_info_out->bits_per_sample == p_info_in->bits_per_sample
646 && p_info_out->channels == p_info_in->channels ){
647 no_conversion = true;
648 LOGI("FormatConverterStream: no conversion")
649 } else if (p_info_out->channels == p_info_in->channels){
650 // copy 1:1
651 convert.multiply = 1;
652 convert.step = 1;
653 LOGI("FormatConverterStream: no channel conversion")
654 } else if (p_info_in->channels == 1 && p_info_out->channels>1){
655 // generate multiple output channels per frame
656 convert.multiply = p_info_out->channels;
657 convert.step = 1;
658 LOGI("FormatConverterStream: multiply channels by %d", convert.multiply);
659
660 } else if (p_info_out->channels == 1){
661 convert.multiply = 1;
662 convert.step = p_info_in->channels; // just output 1 channel per frame
663 LOGI("FormatConverterStream: no conversion 1 channel")
664 } else {
665 convert.multiply = 0;
666 LOGE("invalid input vs output channels");
667 }
668 }
669 }
