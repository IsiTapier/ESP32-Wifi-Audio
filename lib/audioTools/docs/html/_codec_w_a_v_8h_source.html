48 void begin(uint8_t* buffer, size_t len){
49 LOGI("WAVHeader len: %u",(unsigned) len);
50
51 this->buffer = buffer;
52 this->len = len;
53 this->data_pos = 0l;
54
55 memset((void*)&headerInfo, 0, sizeof(WAVAudioInfo));
56 while (!eof()) {
57 uint32_t tag, tag2, length;
58 tag = read_tag();
59 if (eof())
60 break;
61 length = read_int32();
62 if (!length || length >= 0x7fff0000) {
63 headerInfo.is_streamed = true;
64 length = ~0;
65 }
66 if (tag != TAG('R', 'I', 'F', 'F') || length < 4) {
67 seek(length, SEEK_CUR);
68 continue;
69 }
70 tag2 = read_tag();
71 length -= 4;
72 if (tag2 != TAG('W', 'A', 'V', 'E')) {
73 seek(length, SEEK_CUR);
74 continue;
75 }
76 // RIFF chunk found, iterate through it
77 while (length >= 8) {
78 uint32_t subtag, sublength;
79 subtag = read_tag();
80 if (eof())
81 break;
82 sublength = read_int32();
83 length -= 8;
84 if (length < sublength)
85 break;
86 if (subtag == TAG('f', 'm', 't', ' ')) {
87 if (sublength < 16) {
88 // Insufficient data for 'fmt '
89 break;
90 }
91 headerInfo.format = read_int16();
92 headerInfo.channels = read_int16();
93 headerInfo.sample_rate = read_int32();
94 headerInfo.byte_rate = read_int32();
95 headerInfo.block_align = read_int16();
96 headerInfo.bits_per_sample = read_int16();
97 if (headerInfo.format == 0xfffe) {
98 if (sublength < 28) {
99 // Insufficient data for waveformatex
100 break;
101 }
102 skip(8);
103 headerInfo.format = read_int32();
104 skip(sublength - 28);
105 } else {
106 skip(sublength - 16);
107 }
108 headerInfo.is_valid = true;
109 } else if (subtag == TAG('d', 'a', 't', 'a')) {
110 sound_pos = tell();
111 headerInfo.data_length = sublength;
112 if (!headerInfo.data_length || headerInfo.is_streamed) {
113 headerInfo.is_streamed = true;
114 logInfo();
115 return;
116 }
117 seek(sublength, SEEK_CUR);
118 } else {
119 skip(sublength);
120 }
121 length -= sublength;
122 }
123 if (length > 0) {
124 // Bad chunk?
125 seek(length, SEEK_CUR);
126 }
127 }
128 logInfo();
129 return;
130 }
