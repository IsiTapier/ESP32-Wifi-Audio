671 static char* itoa(int n, char s[]) {
672 int i, sign;
673 if ((sign = n) < 0) /* record sign */
674 n = -n; /* make n positive */
675 i = 0;
676 do { /* generate digits in reverse order */
677 s[i++] = n % 10 + '0'; /* get next digit */
678 } while ((n /= 10) > 0); /* delete it */
679 if (sign < 0)
680 s[i++] = '-';
681 s[i] = '\0';
682 reverse(s);
683 return s;
684 }
