648class SilenceRemovalConverter : public BaseConverter<T> {
649 public:
650
651 SilenceRemovalConverter(int n = 8, int aplidudeLimit = 2) {
652 set(n, aplidudeLimit);
653 }
654
655 virtual size_t convert(uint8_t *data, size_t size) override {
656 if (!active) {
657 // no change to the data
658 return size;
659 }
660 size_t sample_count = size / sizeof(T);
661 size_t write_count = 0;
662 T *audio = (T *)data;
663
664 // find relevant data
665 T *p_buffer = (T *)data;
666 for (int j = 0; j < sample_count; j++) {
667 int pos = findLastAudioPos(audio, j);
668 if (pos < n) {
669 write_count++;
670 *p_buffer++ = audio[j];
671 }
672 }
673
674 // write audio data w/o silence
675 size_t write_size = write_count * sizeof(T);
676 LOGI("filtered silence from %d -> %d", (int)size, (int)write_size);
677
678 // number of empty samples of prior buffer
679 priorLastAudioPos = findLastAudioPos(audio, sample_count - 1);
680
681 // return new data size
682 return write_size;
683 }
684
685 protected:
686 bool active = false;
687 const uint8_t *buffer = nullptr;
688 int n;
689 int priorLastAudioPos = 0;
690 int amplidude_limit = 0;
691
692 void set(int n = 5, int aplidudeLimit = 2) {
693 LOGI("begin(n=%d, aplidudeLimit=%d", n, aplidudeLimit);
694 this->n = n;
695 this->amplidude_limit = aplidudeLimit;
696 this->priorLastAudioPos = n+1; // ignore first values
697 this->active = n > 0;
698 }
699
700
701 // find last position which contains audible data
702 int findLastAudioPos(T *audio, int pos) {
703 for (int j = 0; j < n; j++) {
704 // we are before the start of the current buffer
705 if (pos - j <= 0) {
706 return priorLastAudioPos;
707 }
708 // we are in the current buffer
709 if (abs(audio[pos - j]) > amplidude_limit) {
710 return j;
711 }
712 }
713 return n + 1;
