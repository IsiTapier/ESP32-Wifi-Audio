90 inline size_t copy(){
91 LOGD(LOG_METHOD);
92 size_t result = 0;

95 size_t bytes_to_read = buffer_size;
96 size_t bytes_read = 0;
97 int to_write = to->availableForWrite();
98
99 if (len>0){
100 bytes_to_read = min(len, static_cast<size_t>(buffer_size));
101 size_t samples = bytes_to_read / sizeof(T);
102 bytes_to_read = samples * sizeof(T);
103 // don't overflow buffer
104 if (to_write>0){
105 bytes_to_read = min((int)bytes_to_read, to_write);
106 }
107
108 // get the data now
109 bytes_read = from->readBytes((uint8_t*)buffer, bytes_to_read);
110

113 is_first = false;
114

117
118 // callback with unconverted data
119 if (onWrite!=nullptr) onWrite(onWriteObj, buffer, result);
120 #ifndef COPY_LOG_OFF
121 LOGI("StreamCopy::copy %u -> %u -> %u bytes - in %u hops", (unsigned int)bytes_to_read,(unsigned int) bytes_read, (unsigned int)result, (unsigned int)delayCount);

242 if (buffer==nullptr) return 0;
243 size_t total = 0;
244 int retry = 0;
245 while(total<len){
246 size_t written = to->write((const uint8_t*)buffer+total, len-total);
247 total += written;
248 delayCount++;
249
250 if (retry++ > retryLimit){
251 LOGE("write to target has failed!");
252 break;
253 }
254
255 if (retry>1) {
256 delay(5);
257 LOGI("try write - %d ",retry);
258 }
259
260 }
261 return total;
