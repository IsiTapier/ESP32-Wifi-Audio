91 // execute action
92 Action* a = &(actions[pos]);
93 if (a->enabled) {
94 bool value = readValue(a);
95 if (a->actionOn != nullptr && a->actionOff != nullptr) {
96 // we have on and off action defined
97 if (value != a->lastState) {
98 // LOGI("processActions: case with on and off");
99 // execute action -> reports active instead of pin state
100 if ((value && a->activeLogic == ActiveHigh) ||
101 (!value && a->activeLogic == ActiveLow)) {
102 a->actionOn(true, a->pin, a->ref);
103 } else {
104 a->actionOff(false, a->pin, a->ref);
105 }
106 a->lastState = value;
107 }
108 } else if (a->activeLogic == ActiveChange) {
109 bool active = (a->activeLogic == ActiveLow) ? !value : value;
110 // reports pin state
111 if (value != a->lastState) {
112 //LOGI("processActions: ActiveChange");
113 // execute action
114 a->actionOn(active, a->pin, a->ref);
115 a->lastState = value;
116 }
117 } else {
118 bool active = (a->activeLogic == ActiveLow) ? !value : value;
119 if (active &&
120 (active != a->lastState || millis() > a->debounceTimeout)) {
121 //LOGI("processActions: %d Active %d - %d", a->pin, value, digitalRead(a->pin));
122 // execute action
123 a->actionOn(active, a->pin, a->ref);
124 a->lastState = active;
125 a->debounceTimeout = millis() + DEBOUNCE_DELAY;
126 }
127 }
128 }
129 pos++;
130 if (pos >= maxIdx) {
131 pos = 0;
132 }
